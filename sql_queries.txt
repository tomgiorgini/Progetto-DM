CREATE TABLE UFC_Fights (
    Fighter1 VARCHAR(100),
    Fighter2 VARCHAR(100),
    Date DATE,
    Location VARCHAR(100),
    Country VARCHAR(50),
    Winner VARCHAR(10),
    TitleBout BOOLEAN,
    WeightClass VARCHAR(50),
    Gender VARCHAR(10),
    NumberOfRounds INT,
    LoseStreakDif INT,
    WinStreakDif INT,
    LongestWinStreakDif INT,
    WinDif INT,
    LossDif INT,
    TotalRoundDif INT,
    TotalTitleBoutDif INT,
    KODif INT,
    SubDif INT,
    HeightDif FLOAT,
    ReachDif FLOAT,
    AgeDif INT,
    SigStrDif FLOAT,
    AvgSubAttDif FLOAT,
    AvgTDDif FLOAT,
    EmptyArena BOOLEAN,
    BetterRank VARCHAR(10),
    Finish VARCHAR(20),
    FinishDetails VARCHAR(100),
    FinishRound FLOAT,
    FinishRoundTime VARCHAR(10),
    TotalFightTimeSecs FLOAT,
    "1Odds" FLOAT,
    "2Odds" FLOAT,
    "1ExpectedValue" FLOAT,
    "2ExpectedValue" FLOAT,
    "1DecOdds" FLOAT,
    "2DecOdds" FLOAT,
    "1SubOdds" FLOAT,
    "2SubOdds" FLOAT,
    "1KOOdds" FLOAT,
    "2KOOdds" FLOAT,
    PRIMARY KEY (Fighter1, Fighter2, Date)
);

    WinsByDecisionSplit INT,
    WinsByDecisionUnanimous INT,
    WinsByKO INT,
    WinsBySubmission INT,
    WinsByTKODoctorStoppage INT,
    Wins INT,

    ReachCms FLOAT,
