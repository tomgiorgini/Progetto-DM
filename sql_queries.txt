CREATE TABLE UFC_Fights (
    RedFighter VARCHAR(100),
    BlueFighter VARCHAR(100),
    Date DATE,
    Location VARCHAR(100),
    Country VARCHAR(50),
    Winner VARCHAR(10),
    TitleBout BOOLEAN,
    WeightClass VARCHAR(30),
    Gender VARCHAR(10),
    NumberOfRounds INT,
    
    LoseStreakDif INT,
    WinStreakDif INT,
    LongestWinStreakDif INT,
    WinDif INT,
    LossDif INT,
    TotalRoundDif INT,
    TotalTitleBoutDif INT,
    KODif INT,
    SubDif INT,
    HeightDif FLOAT,
    ReachDif FLOAT,
    AgeDif INT,
    SigStrDif FLOAT,
    AvgSubAttDif FLOAT,
    AvgTDDif FLOAT,

    EmptyArena BOOLEAN,
    BetterRank VARCHAR(10),

    Finish VARCHAR(10),
    FinishDetails VARCHAR(100),
    FinishRound FLOAT,
    FinishRoundTime VARCHAR(10),
    TotalFightTimeSecs FLOAT,

    RedOdds FLOAT,
    BlueOdds FLOAT,
    RedExpectedValue FLOAT,
    BlueExpectedValue FLOAT,
    RedDecOdds FLOAT,
    BlueDecOdds FLOAT,
    RSubOdds FLOAT,
    BSubOdds FLOAT,
    RKOOdds FLOAT,
    BKOOdds FLOAT,

    PRIMARY KEY (RedFighter, BlueFighter, Date)
);

    TotalRoundsFought INT,

    WinsByDecisionMajority INT,
    WinsByDecisionSplit INT,
    WinsByDecisionUnanimous INT,
    WinsByKO INT,
    WinsBySubmission INT,
    WinsByTKODoctorStoppage INT,

    Stance VARCHAR(20),

    ReachCms FLOAT,

    WFlyweightRank INT,
